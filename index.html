Hey, so I figured out a simple function to calculate the sum of all even numbers in an array. Here it is:

            
            function sumEvenNumbers(arr) {
            let sum = 0;
            for (let num of arr) {
            if (num % 2 === 0) {
            sum += num;
            }
            }
            return sum;
            }
            
        
            
            console.log(sumEvenNumbers([1, 2, 3, 4, 5])); // ( 2 + 4 ) : Output: 6
            console.log(sumEvenNumbers([10, 20, 30, 40])); // (10 + 20 + 30 + 40 ) : Output: 100
